# The number of qubits being generated to make the key
key_len = 10

# Both Alice and Bob must run this code to ensure their qubits are encoded
def encoding(length): 
    
    # Two lists are required: one to keep and one to send to the other person
    list1 = []
    list2 = []
    
    # For the length 
    for i in range(length):
        
        # Create a brand new quantum circuit called qc. 
        # Remember that the qubit will be in state |0> by default
        qc1 = q.QuantumCircuit(1,1)
        qc2 = q.QuantumCircuit(1,1)
        
        # Create random number to create random states
        rand_num = randrange(3)
        
        if rand_num == 0:
            # We want to encode a |0> state, as states are intialized
            # in |0> by default we don't need to add anything here
            pass
            
        elif rand_num == 1:
                # We want to encode a |1> state
                # We apply an X gate to generate |1>
                qc1.x(0)   
                qc2.x(0)
                
        elif rand_num == 2:
                # We apply an H gate to generate |+>
                qc1.h(0)
                qc2.h(0)
                
        elif rand_num == 3:
                # We apply an X and an H gate to generate |->
                qc1.x(0)
                qc1.h(0)
                qc2.x(0)
                qc2.x(0)
        
        # Add the qubit to the lists
        list1.append(qc1)
        list2.append(qc2)
    
    # Return both identical lists
    return list1, list2

# Calling the method to run store both the lists in the two variables: qubit_list1 and qubit_list2
qubit_list1, qubit_list2 = encoding(key_len)
